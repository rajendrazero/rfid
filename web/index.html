Tujuan & Requirements

Tujuan Aplikasi

Membangun sistem manajemen troli bandara berbasis RFID untuk meningkatkan efisiensi pelacakan dan manajemen troli di lingkungan bandara.

  • Mengurangi waktu pencarian troli oleh petugas
  • Memantau ketersediaan troli di berbagai lokasi bandara
  • Memberikan data real-time tentang status troli
  • Meningkatkan pengalaman pengguna dengan troli yang mudah ditemukan
  • Memudahkan manajemen inventaris troli

Functional Requirements

  1. Sistem login dengan role user dan admin
  2. Pelacakan lokasi troli melalui RFID
  3. Fitur CRUD (Create, Read, Update, Delete) untuk troli
  4. Pemberian status troli (tersedia, rusak, dalam perbaikan)
  5. Monitoring user aktif dan aktivitas mereka
  6. Dashboard informasi troli dengan detail lengkap
  7. Laporan historis pergerakan troli

Script Pembuatan Tabel

CREATE TABLE public.locations (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  name character varying NOT NULL,
  description text,
  created_at timestamp with time zone DEFAULT now(),
  CONSTRAINT locations_pkey PRIMARY KEY (id)
);

CREATE TABLE public.roles (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  name character varying NOT NULL UNIQUE,
  CONSTRAINT roles_pkey PRIMARY KEY (id)
);

CREATE TABLE public.trolley_location_logs (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  trolley_id uuid NOT NULL,
  location_id uuid NOT NULL,
  detected_at timestamp with time zone NOT NULL DEFAULT now(),
  detected_by uuid,
  CONSTRAINT trolley_location_logs_pkey PRIMARY KEY (id),
  CONSTRAINT trolley_location_logs_trolley_id_fkey FOREIGN KEY (trolley_id) REFERENCES public.trolleys(id),
  CONSTRAINT trolley_location_logs_location_id_fkey FOREIGN KEY (location_id) REFERENCES public.locations(id),
  CONSTRAINT trolley_location_logs_detected_by_fkey FOREIGN KEY (detected_by) REFERENCES public.users(id)
);

CREATE TABLE public.trolley_statuses (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  name character varying NOT NULL,
  description text,
  CONSTRAINT trolley_statuses_pkey PRIMARY KEY (id)
);

CREATE TABLE public.trolleys (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  rfid_code character varying NOT NULL UNIQUE,
  trolley_code character varying NOT NULL UNIQUE,
  status_id uuid,
  created_at timestamp with time zone DEFAULT now(),
  updated_at timestamp without time zone,
  deleted_at timestamp without time zone,
  CONSTRAINT trolleys_pkey PRIMARY KEY (id),
  CONSTRAINT trolleys_status_id_fkey FOREIGN KEY (status_id) REFERENCES public.trolley_statuses(id)
);

CREATE TABLE public.users (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  username character varying NOT NULL UNIQUE,
  password text NOT NULL,
  fullname character varying,
  email character varying UNIQUE,
  role_id uuid,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  updated_at timestamp without time zone,
  deleted_at timestamp without time zone,
  is_logged_in boolean DEFAULT false,
  last_login timestamp with time zone,
  CONSTRAINT users_pkey PRIMARY KEY (id),
  CONSTRAINT users_role_id_fkey FOREIGN KEY (role_id) REFERENCES public.roles(id)
);
